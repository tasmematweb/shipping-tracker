import 'dart:convert'; import 'package:crypto/crypto.dart'; import 'package:http/http.dart' as http; class JnTTrackingService { // بيانات الحساب الخاصة بك final String _apiAccount = "292508153084379141"; final String _privateKey = "a0a1047cce70493c9d5d29704f05d0d9"; // رابط الـ API للإنتاج (Production Endpoint) final String _baseUrl = "https://open.jtjms-eg.com/web-openapi/api/logistics/trace"; /// دالة جلب تتبع الشحنة مع التعامل مع التوقيع (Digest) Future<Map<String, dynamic>?> fetchTrackingDetails(String billCode) async { try { // 1. توليد الـ timestamp الحالي final String timestamp = DateTime.now().millisecondsSinceEpoch.toString(); // 2. تجهيز الـ bizContent final Map<String, dynamic> bizContentMap = { "billCodes": [billCode] }; final String bizContent = jsonEncode(bizContentMap); // 3. حساب الـ Digest final String rawData = bizContent + timestamp + _privateKey; final bytes = utf8.encode(rawData); final md5Hash = md5.convert(bytes); final String digest = base64.encode(md5Hash.bytes); // 4. إعداد الـ Headers final Map<String, String> headers = { "Content-Type": "application/x-www-form-urlencoded; charset=UTF-8", "apiAccount": _apiAccount, "timestamp": timestamp, "digest": digest, }; // 5. إعداد الـ Body مع ترميز URL final Map<String, String> body = { "bizContent": Uri.encodeComponent(bizContent), }; // 6. تنفيذ الطلب (POST Request) final response = await http.post( Uri.parse(_baseUrl), headers: headers, body: body, ); if (response.statusCode == 200) { final data = jsonDecode(response.body); if (data["code"] == "1" || data["code"] == 1 || data["msg"] == "success") { return data; } else { print("API Error Code: ${data["code"]}, Message: ${data["msg"]}"); return null; } } else { print("HTTP Error: ${response.statusCode}"); print("Response Body: ${response.body}"); return null; } } catch (e) { print("Exception occurred: $e"); return null; } } /// دالة مساعدة لترجمة وتحويل رموز الـ Scan Types Map<String, String> parseScanDetails(Map<String, dynamic> detailItem) { String scanTypeName = detailItem["scanTypeName"] ?? ""; String desc = detailItem["desc"] ?? ""; String problemDescription = detailItem["problemDescription"] ?? ""; final Map<String, String> commonScanNames = { "Pick up": "تم الاستلام (جمع الشحنة)", "Enter branch Scan": "مسح دخول الفرع", "Sending scan": "مسح الإرسال", "Arrival Scan": "مسح الوصول", "OFD Scan": "جاري توصيل الشحنة (متجه للعميل)", "Sign scan": "تم التوقيع والتسليم بنجاح", "Problem Shipment Scan": "وجد مشكلة في الشحنة (Abnormal)", "Return scan": "مسح الإرجاع", }; String arabicScanName = commonScanNames[scanTypeName] ?? scanTypeName; return { "scanName": arabicScanName, "description": desc, "problemReason": problemDescription, }; } }
